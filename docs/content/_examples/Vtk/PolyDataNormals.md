import QtQuick 2.6

import Vtk 1.0 as Vtk

    id: root;

    anchors.fill: parent;

    Vtk.Viewer {

        mouseEnabled: true;

        Vtk.Actor {

            from: filter; bind: "featureAngle";
            min: 0; max: 180; step: 1; value: 15
        }

        Utils.CheckBox {
            from: filter; bind: "splitting";
        }

            from: filter; bind: "consistency";
        }

            from: filter; bind: "autoOrientNormals";
